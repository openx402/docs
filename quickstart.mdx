## Overview

This guide will walk you through integrating OpenX402 into your application in under 10 minutes. By the end, you'll have a working endpoint that accepts micropayments.

**Facilitator URL**: `https://open.x402.host`

## Prerequisites

- A wallet with some USDC on Base (for testing)
- Basic knowledge of Node.js or Python
- An API or service you want to monetize

## Step 1: Test the Facilitator

First, let's verify the facilitator is accessible and see what networks it supports.

```bash
curl https://open.x402.host/supported
```

You should see a response listing supported payment networks:

```json
{
  "networks": ["base", "xlayer"],
  "facilitator": "openx402",
  "version": "1.0"
}
```

## Step 2: Create a Simple Protected Endpoint

Let's create a basic server with an endpoint that requires payment.

<CodeGroup>
```javascript Node.js (Express)
const express = require('express');
const app = express();

app.use(express.json());

// Unprotected endpoint
app.get('/free', (req, res) => {
  res.json({ message: 'This is free!' });
});

// Protected endpoint
app.post('/premium', async (req, res) => {
  const paymentProof = req.headers['x-402-payment'];

  if (!paymentProof) {
    return res.status(402).json({
      error: 'Payment required',
      facilitator: 'https://open.x402.host'
    });
  }

  // Verify payment with OpenX402
  const verified = await verifyPayment(paymentProof);

  if (verified) {
    res.json({ data: 'Your premium content here!' });
  } else {
    res.status(402).json({ error: 'Invalid payment' });
  }
});

app.listen(3000, () => {
  console.log('Server running on port 3000');
});
```

```python Python (Flask)
from flask import Flask, request, jsonify
import requests

app = Flask(__name__)

@app.route('/free')
def free_endpoint():
    return jsonify({"message": "This is free!"})

@app.route('/premium', methods=['POST'])
def premium_endpoint():
    payment_proof = request.headers.get('X-402-Payment')

    if not payment_proof:
        return jsonify({
            "error": "Payment required",
            "facilitator": "https://open.x402.host"
        }), 402

    # Verify payment with OpenX402
    verified = verify_payment(payment_proof)

    if verified:
        return jsonify({"data": "Your premium content here!"})
    else:
        return jsonify({"error": "Invalid payment"}), 402

if __name__ == '__main__':
    app.run(port=3000)
```
</CodeGroup>

## Step 3: Implement Payment Verification

Now implement the `verifyPayment` function that calls OpenX402's `/verify` endpoint.

<CodeGroup>
```javascript Node.js
async function verifyPayment(paymentProof) {
  try {
    const response = await fetch('https://open.x402.host/verify', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        proof: paymentProof,
        network: 'base'
      })
    });

    const result = await response.json();
    return result.valid === true;
  } catch (error) {
    console.error('Verification error:', error);
    return false;
  }
}
```

```python Python
def verify_payment(payment_proof):
    try:
        response = requests.post(
            'https://open.x402.host/verify',
            json={
                'proof': payment_proof,
                'network': 'base'
            }
        )
        result = response.json()
        return result.get('valid') == True
    except Exception as e:
        print(f'Verification error: {e}')
        return False
```
</CodeGroup>

## Step 5: Add Settlement (Optional)

For production, you may want to explicitly settle transactions:

```javascript
async function settlePayment(transactionId) {
  const response = await fetch('https://open.x402.host/settle', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      transactionId: transactionId,
      network: 'base'
    })
  });

  return await response.json();
}

// Call after delivering the service
app.post('/premium', async (req, res) => {
  // ... verification code ...

  // Deliver service
  res.json({ data: 'Premium content' });

  // Settle in background
  await settlePayment(paymentProof.txId);
});
```

## Complete Example

Here's a full working example:

```javascript
const express = require('express');
const fetch = require('node-fetch');
const app = express();

app.use(express.json());

const FACILITATOR = 'https://open.x402.host';

async function verifyWithOpenX402(proof) {
  const response = await fetch(`${FACILITATOR}/verify`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ proof, network: 'base' })
  });
  const result = await response.json();
  return result.valid;
}

app.post('/api/data', async (req, res) => {
  const proof = req.headers['x-402-payment'];

  if (!proof) {
    return res.status(402).json({
      error: 'Payment required',
      facilitator: FACILITATOR
    });
  }

  const valid = await verifyWithOpenX402(proof);

  if (!valid) {
    return res.status(402).json({ error: 'Invalid payment' });
  }

  // Payment verified - deliver service
  res.json({
    data: 'Your premium data here',
    timestamp: Date.now()
  });
});

app.listen(3000);
```
